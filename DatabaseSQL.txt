create table packages (
    id bigint generated by default as identity,
    creator_username varchar references users(username),
    creation_date varchar not null,
    sender_name varchar not null,
    sender_phone varchar not null,
    sender_department varchar references departments(number),
    package_type varchar not null,
    package_description varchar not null,
    package_price varchar not null,
    package_params varchar not null,
    recipient_name varchar not null,
    recipient_phone varchar not null,
    recipient_department varchar references departments(number),
    status varchar not null
);

/// -----------------------------------------------

create table departments
(
    id integer generated by default as identity primary key,
    number varchar(255) not null unique,
    address varchar(255) not null unique,
    type varchar(255) not null,
    weight_restriction varchar(255) not null,
    dim_restriction_send varchar(255) not null,
    dim_restriction_receive varchar(255) not null,
    services varchar(255) not null,
    working_hours integer references workinghours(id)
);

/// -----------------------------------------------

create table users
(
    id       integer generated by default as identity
        constraint users_pk
            primary key,
    username varchar(255) not null
        unique,
    password varchar(255) not null,
    name     varchar(255) not null,
    email    varchar(255) not null
        unique,
    phone    varchar(255) not null
        unique,
    role     varchar(255) not null
);

alter table public.users
    owner to postgres;

/// -----------------------------------------------

create table public.workinghours
(
    id                    integer generated by default as identity
        primary key,
    mon                   varchar(255),
    mon_same_day_shipping varchar(255),
    mon_shipment_arrival  varchar(255),
    tue                   varchar(255),
    tue_same_day_shipping varchar(255),
    tue_shipment_arrival  varchar(255),
    wed                   varchar(255),
    wed_same_day_shipping varchar(255),
    wed_shipment_arrival  varchar(255),
    thu                   varchar(255),
    thu_same_day_shipping varchar(255),
    thu_shipment_arrival  varchar(255),
    fri                   varchar(255),
    fri_same_day_shipping varchar(255),
    fri_shipment_arrival  varchar(255),
    sat                   varchar(255),
    sat_same_day_shipping varchar(255),
    sat_shipment_arrival  varchar(255),
    sun                   varchar(255),
    sun_same_day_shipping varchar(255),
    sun_shipment_arrival  varchar(255)
);

alter table public.workinghours
    owner to postgres;
